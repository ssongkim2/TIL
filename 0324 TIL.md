from itertools import combinations

def solution(N, M):
    num1 = []
    num2 = []
    sum1 = 0
    sum2 = 0
    max = 0
    for row in range(N):
        for col in range(N-M+1):
            num1 = numbers[row][col:col+M]            #여기서 잘라서 들어가는게 어차피 리스트
            if (col + M) > N:                         #애시당초 col+M인 경우 pass하고 내려보냄
                continue
            else:
                for row2 in range(row, N):            #일단 시작 row에서 시작
                    if row2 == row:
                        A = col + 1                   #같은줄 일때
                    else:
                        A = 0                         #다른줄일때
                        for col2 in range(A, N-M+1):
                            num2 = numbers[row2][col2:col2+M]                      #여기서 실수함..... 인덱스 잘보자!
                            max1 = 0
                            max2 = 0
                            for K in range(0, M+1):
                                result1 = list(combinations(num1, K))
                                result2 = list(combinations(num2, K))
                                                                                    #부분집합 다 구하기

                                for i1 in result1:                                   #result1 안에 값들
                                    sum1 = sum(i1)
                                    total1 = 0
                                    if sum1 > C:
                                        continue
                                    else:
                                        for d1 in i1:
                                            total1 += d1 ** 2
                                            if total1 > max1:
                                                max1 = total1
                                            else:
                                                continue
                                for i2 in result2:
                                    sum2 = sum(i2)
                                    total2 = 0
                                    if sum2 > C:
                                        continue
                                    else:
                                        for d2 in i2:
                                            total2 += d2 ** 2
                                            if total2 > max2:
                                                max2 = total2
                                            else:
                                                continue                                  #아래코드들은 최댓값 찾기
                            if max1 + max2 > max:
                                max = max1 + max2
    return max


                                                           #부분집합의 인자의 합이 C보다 크면
#이제 구현할 것 부분집합으로 받아서(조합으로 1에서 M까지) 늘려가며 비교 하고 부분집합 안에 값이 K이상이면 skip
#이하이면 최대값이랑 비교하면서 keep going
